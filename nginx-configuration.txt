user nginx;
worker_processes auto;
error_log /var/log/nginx/error.log;
pid /run/nginx.pid;

# Load dynamic modules. See /usr/share/doc/nginx/README.dynamic.include /usr/share/nginx/modules/*.conf;

events {
    worker_connections 1024;
}

http {
    log_format  main  '$remote_addr - $remote_user [$time_local] "$request" '
                      '$status $body_bytes_sent "$http_referer" '
                      '"$http_user_agent" "$http_x_forwarded_for"';

    access_log  /var/log/nginx/access.log  main;

    sendfile            on;
    tcp_nopush          on;
    tcp_nodelay         on;
    keepalive_timeout   65;
    types_hash_max_size 4096;

    include             /etc/nginx/mime.types;
    default_type        application/octet-stream;

    # Load modular configuration files from the /etc/nginx/conf.d directory.
    # See http://nginx.org/en/docs/ngx_core_module.html#include
    # for more information.
#    include /etc/nginx/conf.d/*.conf;


#START SERVICE NAME


    server {
        listen       80;
        listen       [::]:80;
        server_name  exampledomain.com;
        root         /usr/share/nginx/html;

        return 301 https://$host;

        # Load configuration files for the default server block.
#        include /etc/nginx/default.d/*.conf;

        error_page 404 /404.html;
        location = /404.html {
        }

        error_page 500 502 503 504 /50x.html;
        location = /50x.html {
        }
    }

# Settings for a TLS enabled server.



    server {
        listen       443 ssl http2;
        listen       [::]:443 ssl http2;
        server_name  exampledomain.com;
       root         /usr/share/nginx/html;

	#SSL Configuration
        ssl_certificate "/etc/nginx/path/example-ssl-bundle.crt";
        ssl_certificate_key "/etc/nginx/path/exampledomain.com.key";
        ssl_session_cache shared:SSL:1m;
        ssl_session_timeout  10m;

        # ssl_ciphers PROFILE=SYSTEM;
        ssl_prefer_server_ciphers on;
        keepalive_timeout 70;

#        # Load configuration files for the default server block.
 #       include /etc/nginx/default.d/*.conf;

        #configure the port redirect
        location / {
          proxy_pass "http://127.0.0.1:1000";
          proxy_hide_header X-Powered-By;
          proxy_hide_header Server;
        }

        error_page 404 /404.html;
            location = /40x.html {
        }
